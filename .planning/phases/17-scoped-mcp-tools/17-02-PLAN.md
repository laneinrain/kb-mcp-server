  ```typescript
  export function getToolAllowedDocumentIds(): ReadonlySet<string> | undefined {
    const ctx = mcpCallerStorage.getStore();
    if (!ctx || ctx.authMode !== "user") {
      return undefined;
    }
    return ctx.allowedDocumentIds;
  }
  ```
